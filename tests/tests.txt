
PHONE_IP=172.16.1.52
## INDI tests.

# 1. Discovery — list all devices and properties                                                                                               
python tests/indi_test_client.py --host $PHONE_IP                                                                                            
                                                                                                                                                 
# 2. Connect to default CCD device                                                                                                             
python tests/indi_test_client.py --host $PHONE_IP --connect                                                                                  
                                                                                                                                                 
# 3. Lens info — print focal length and aperture for all CCD devices                                                                           
python tests/indi_test_client.py --host $PHONE_IP --lens-info
                                                                                                                                                 
# 4. Focuser info — print position, range, direction                                                                                           
python tests/indi_test_client.py --host $PHONE_IP --focus-info
                                                                                                                                                 
# 5. Focuser absolute move                                                                                                                     
python tests/indi_test_client.py --host $PHONE_IP --focus-abs 500
python tests/indi_test_client.py --host $PHONE_IP --focus-abs 0                                                                              
                                                                                                                                                 
# 6. Focuser relative move                                                                                                                     
python tests/indi_test_client.py --host $PHONE_IP --focus-rel 100                                                                            
                                                                                                                                                 
# 7. Capture 1s exposure (default camera)
python tests/indi_test_client.py --host $PHONE_IP --device "PocketScope Main" --capture 1.0 -o indi_capture.fits
                                                                                                                                                 
# 8. Capture from a specific camera device                                                                                                     
python tests/indi_test_client.py --host $PHONE_IP --device "PocketScope Main" --capture 1.0 -o indi_main.fits                                
                                                                                                                                                 
# 9. Capture with longer exposure                  
python tests/indi_test_client.py --host $PHONE_IP --capture 5.0 -o indi_5s.fits                                                              
                                                                                                                                                 
# 10. Combined: connect, focus, then capture                                                                                                   
python tests/indi_test_client.py --host $PHONE_IP --focus-abs 500 --capture 2.0 -o indi_focused.fits
